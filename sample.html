$('#results_scroll .grid-cell').live('mouseover', function(e) {

tfmap_trail_highlight(map, features[0]);
}
} else if (type === 'region') {

map.setFeatureState({source: 'regions', id: nid}, {hover: true});

} else if (type === 'route') {

}

function updateURLswithParams() {
urlParams = {};
urlParams.lat = map.getCenter().lat.toFixed(6);
urlParams.lon = map.getCenter().lng.toFixed(6);
urlParams.z = map.getZoom();
if (jQuery('#winter_checkbox').is(':checked')) urlParams.season = 2;

var activitytype = $('#activitytype').val();
if (activitytype) {
urlParams.activitytype = activitytype;
}

$('#recommendedLink').attr("href", url+'?'+query);
}
